    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css" />
    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
    <style>
        body { margin: 0; padding: 0; }
        #map { width: 100%; height: 100vh; }
        .location-btn {
            position: absolute; top: 10px; right: 10px; z-index: 1000;
            padding: 10px; background: white; border: 2px solid #3388ff; cursor: pointer; border-radius: 5px;
        }
    </style>

<body>

<div id="map"></div>
<button class="location-btn" onclick="startTracking()">내 위치 찾기</button>

<script>
    // 1. 지도 초기화 (기본 중심: 서울)
    var map = L.map('map').setView([37.5665, 126.9780], 13);

    // 2. 위성 배경지도 (Esri World Imagery)
    L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', {
        attribution: 'Tiles &copy; Esri'
    }).addTo(map);

    // 3. 레이어 스타일 설정 함수
    function styleLayer(color) {
        return { color: color, weight: 3, opacity: 0.8, fillOpacity: 0.2 };
    }

    // 4. 가공.geojson 로드
    fetch('가공.geojson')
        .then(res => res.json())
        .then(data => {
            console.log("가공.geojson 로드 완료");
            var layer = L.geoJSON(data, { style: styleLayer("#ff0000") }).addTo(map);
            // 데이터가 있는 곳으로 지도 이동
            if (data.features.length > 0) map.fitBounds(layer.getBounds());
        })
        .catch(err => alert("가공.geojson 파일을 찾을 수 없거나 보안 문제입니다. 서버를 통해 실행하세요."));

    // 5. 조사시점.geojson 로드
    fetch('조사시점.geojson')
        .then(res => res.json())
        .then(data => {
            console.log("조사시점.geojson 로드 완료");
            L.geoJSON(data, { style: styleLayer("#0000ff") }).addTo(map);
        })
        .catch(err => console.error("조사시점 로드 실패"));

    // 6. 위치 추적 로직
    var userMarker;
    function onLocationFound(e) {
        if (userMarker) map.removeLayer(userMarker);
        userMarker = L.circleMarker(e.latlng, {radius: 8, color: 'white', fillColor: '#3388ff', fillOpacity: 1, weight: 2}).addTo(map);
        map.setView(e.latlng, 16);
    }

    function onLocationError(e) {
        alert("위치 정보를 가져올 수 없습니다: " + e.message);
    }

    map.on('locationfound', onLocationFound);
    map.on('locationerror', onLocationError);

    function startTracking() {
        map.locate({setView: true, maxZoom: 16, enableHighAccuracy: true});
    }
</script>
</body>
