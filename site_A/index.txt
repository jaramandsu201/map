            position: absolute; top: 10px; right: 10px; z-index: 1000;

<div id="map"></div>
<div class="btn-container">
    <button onclick="startTracking()">📍 내 위치 추적</button>
</div>

<script>
    // 1. 지도 초기화 (초기 위치는 대략적인 한국 중심)
    var map = L.map('map').setView([36.5, 127.5], 7);

    // 2. 최신 위성 영상 (Esri)
    L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', {
        attribution: 'Tiles &copy; Esri'
    }).addTo(map);

    // 3. 가공.geojson 불러오기
    fetch('./가공.geojson')
        .then(res => res.json())
        .then(data => {
            var layer1 = L.geoJSON(data, { 
                style: { color: "#ff0000", weight: 3, opacity: 0.8 } 
            }).addTo(map);
            // 데이터가 로드되면 해당 위치로 지도 자동 이동
            map.fitBounds(layer1.getBounds());
        })
        .catch(err => console.error("가공 레이어 로드 실패:", err));

    // 4. 조사시점.geojson 불러오기
    fetch('./조사시점.geojson')
        .then(res => res.json())
        .then(data => {
            L.geoJSON(data, { 
                style: { color: "#0000ff", weight: 3, opacity: 0.8 } 
            }).addTo(map);
        })
        .catch(err => console.error("조사시점 레이어 로드 실패:", err));

    // 5. 실시간 위치 추적 (HTTPS인 깃허브에서 작동함)
    var userMarker, userCircle;

    function onLocationFound(e) {
        if (userMarker) {
            userMarker.setLatLng(e.latlng);
            userCircle.setLatLng(e.latlng).setRadius(e.accuracy / 2);
        } else {
            userMarker = L.circleMarker(e.latlng, {radius: 8, color: 'white', fillColor: '#3388ff', fillOpacity: 1}).addTo(map);
            userCircle = L.circle(e.latlng, e.accuracy / 2).addTo(map);
        }
    }

    map.on('locationfound', onLocationFound);
    map.on('locationerror', (e) => alert("위치 접근을 허용해주세요: " + e.message));

    function startTracking() {
        map.locate({watch: true, enableHighAccuracy: true});
    }
</script>
